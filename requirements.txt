# --- Media Downloader (/yt, /yta) ---
# Pinned to a recent release — YouTube/Instagram break old yt-dlp versions
# often, and an unpinned "yt-dlp" can silently stay on a stale cached
# install across redeploys on Render/Railway. Bump this periodically, or
# run `pip install -U yt-dlp` on the host directly if downloads start
# failing again.
# [default] extra bundles yt-dlp-ejs — the external-JS-challenge-solver
# component yt-dlp has required for full YouTube support since 2025.11.12.
# Without it, extraction can still succeed but the actual googlevideo
# download 403s (signature/n-param never gets solved) — see Akbots/ytdl.py's
# js_runtimes opt for the runtime side of this same requirement.
yt-dlp[default]>=2026.07.04

# --- YouTube PO Token auto-generation (fixes "Sign in to confirm you're
# not a bot" / 403 errors without needing manual cookies) ---
# Registers itself as a yt-dlp plugin automatically on install; no extra
# code wiring needed. Falls back silently to cookies if unavailable.
bgutil-ytdlp-pot-provider>=1.2.0

# --- Headless-browser video discovery fallback (JS-rendered players yt-dlp
# can't parse from raw HTML) — used by Akbots/headless.py ---
# `pip install playwright` alone does NOT ship the browser binary — it
# needs a separate `playwright install chromium` step. That's now handled
# automatically:
#   - Docker deploys: the Dockerfile runs it at build time.
#   - Non-Docker (Procfile/buildpack) deploys: Akbots/headless.py
#     self-installs it on first use (one-time ~30-90s delay, cached after).
# Either way, no manual step needed. If it can't install for some reason,
# this fallback just no-ops and the bot falls through to gallery-dl /
# raw-file download as before.
playwright
